21def _pack_envelope_ciphertext(encrypted_dek: bytes, encrypted_data: bytes) -> bytes: 

22 """Packs encrypted DEK and data into combined ciphertext. 

23 

24 Format: [4 bytes: encrypted_dek_length][encrypted_dek][encrypted_data] 

25 """ 

26 encrypted_dek_length = len(encrypted_dek) 

27 return struct.pack("<I", encrypted_dek_length) + encrypted_dek + encrypted_data

30def _unpack_envelope_ciphertext(ct: bytes) -> tuple[bytes, bytes]: 

31 """Unpacks combined ciphertext into encrypted DEK and encrypted data. 

32 

33 Returns tuple of (encrypted_dek, encrypted_data) 

34 """ 

35 if len(ct) < 4: 

36 raise ValueError("Ciphertext too short") 

37 

38 # Extract encrypted DEK length 

39 encrypted_dek_length = struct.unpack("<I", ct[:4])[0] 

40 

41 encrypted_data_offset = 4 + encrypted_dek_length 

42 if len(ct) < encrypted_data_offset: 

43 raise ValueError("Ciphertext format invalid") 

44 

45 # Extract encrypted DEK and encrypted data 

46 encrypted_dek = ct[4:encrypted_data_offset] 

47 encrypted_data = ct[encrypted_data_offset:] 

48 

49 return encrypted_dek, encrypted_data

88class GcpKmsProvider(Provider): 

89 """Implements envelope encryption using Google Cloud KMS and pynacl AEAD. 

90 

91 This provider generates a random data encryption key (DEK), encrypts the plaintext 

92 with the DEK using nacl.secret.Aead, then encrypts the DEK using GCP KMS. 

93 The final ciphertext contains both the encrypted DEK and the encrypted data. 

94 """ 

95 

96 def __init__(self, *, key_name: str, credentials: Credentials): 

97 """Constructs a GcpKmsProvider. 

98 

99 :param key_name: GCP KMS key name (e.g. projects/.../locations/.../keyRings/.../cryptoKeys/...) 

100 :param credentials: GCP service account credentials 

101 """ 

102 self.key_name = key_name 

103 self.kms_client = kms.KeyManagementServiceClient(credentials=credentials) 

104 

105 def name(self) -> str: 

106 return NAME 

107 

108 def encrypt(self, pt: bytes, aad: bytes) -> bytes: 

109 """Encrypts plaintext using envelope encryption.""" 

110 dek = nacl.utils.random(nacl.secret.Aead.KEY_SIZE) 

111 aead = nacl.secret.Aead(dek) 

112 encrypted_data = aead.encrypt(pt, aad) 

113 

114 request = kms.EncryptRequest(name=self.key_name, plaintext=dek) 

115 response = self.kms_client.encrypt(request) 

116 encrypted_dek = response.ciphertext 

117 

118 return _pack_envelope_ciphertext(encrypted_dek, encrypted_data) 

119 

120 def decrypt(self, ct: bytes, aad: bytes) -> bytes: 

121 """Decrypts ciphertext using envelope encryption.""" 

122 encrypted_dek, encrypted_data = _unpack_envelope_ciphertext(ct) 

123 

124 request = kms.DecryptRequest(name=self.key_name, ciphertext=encrypted_dek) 

125 response = self.kms_client.decrypt(request) 

126 dek = response.plaintext 

127 

128 aead = nacl.secret.Aead(dek) 

129 return aead.decrypt(encrypted_data, aad)
