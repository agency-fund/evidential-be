12def _icc_one_way_random_intercept(df: pd.DataFrame, *, cluster_column: str, outcome_column: str) -> float: 

13 """ 

14 ICC for a single-intercept random-cluster model (outcome ~ 1 + (1|cluster)). 

15 

16 Estimated from one-way random-effects ANOVA mean squares (Method of Moments): 

17 

18 ICC = σ²_between / (σ²_between + σ²_within) = (MSB - MSW) / (MSB + (n0 - 1) * MSW) 

19 

20 where n0 is the adjustment for unequal cluster sizes. ICC estimates are clamped to [0, 1]. 

21 """ 

22 # 1. cluster stats 

23 n_total = len(df) 

24 y = df[outcome_column].to_numpy(dtype=np.float64) 

25 codes, _ = pd.factorize(df[cluster_column]) # indices of cluster membership 

26 cluster_sizes = np.bincount(codes).astype(np.float64) 

27 k = len(cluster_sizes) 

28 

29 # 2. ANOVA components for ICC 

30 df_b = k - 1 

31 df_w = n_total - k 

32 if df_w < 1: 

33 raise ValueError("Insufficient within-cluster data (need N > clusters)") 

34 

35 cluster_sums = np.bincount(codes, weights=y) 

36 cluster_means = cluster_sums / cluster_sizes 

37 grand_mean = np.mean(y) 

38 # Sum of Squares Between (SSB) 

39 ssb = np.sum(cluster_sizes * (cluster_means - grand_mean) ** 2) 

40 # Sum of Squares Within (SSW) - subtract the mean of each cluster from each observation 

41 ssw = np.sum((y - cluster_means[codes]) ** 2) 

42 # Derive the mean squares 

43 msb = ssb / df_b 

44 msw = ssw / df_w 

45 

46 # 3. n0 calculation; can think of it as an "effective cluster size" when we have unequal sizes 

47 # for an unbiased estimate of ICC. n0 equals the common cluster size when sizes are balanced. 

48 sum_n2 = np.sum(cluster_sizes**2) 

49 n0 = (n_total - (sum_n2 / n_total)) / df_b 

50 

51 # 4. Final calculation of ICC as the ratio of between-cluster variance to total variance. 

52 # If MSB < MSW, the point estimate for sigma_b is 0 (prevents negative ICC) 

53 var_between = max(0.0, (msb - msw) / n0) 

54 var_within = msw 

55 total_var = var_between + var_within 

56 if total_var == 0: 

57 return 0.0 

58 

59 return float(np.clip(var_between / total_var, 0.0, 1.0))

62def calculate_icc_from_dataframe( 

63 df: pd.DataFrame, 

64 *, 

65 cluster_column: str, 

66 outcome_column: str, 

67) -> float: 

68 """ 

69 Calculate intraclass correlation (ICC) from a dataframe. 

70 

71 Uses a one-way random-effects ANOVA estimator, equivalent to the variance-component 

72 ICC from a linear mixed model with a random intercept per cluster and no fixed 

73 effects beyond the grand mean. 

74 

75 Args: 

76 df: DataFrame containing cluster and outcome columns 

77 cluster_column: Name of the column with cluster identifiers 

78 outcome_column: Name of the column with numeric outcomes 

79 

80 Returns: 

81 ICC value between 0 and 1 

82 

83 Raises: 

84 ValueError: If dataframe is empty or has insufficient data 

85 """ 

86 if len(df) == 0: 

87 raise ValueError("Cannot calculate ICC from empty dataframe") 

88 

89 missing = [name for name in (cluster_column, outcome_column) if name not in df.columns] 

90 if missing: 

91 raise ValueError(f"DataFrame is missing columns: {', '.join(missing)}") 

92 if df[cluster_column].nunique() < 2: 

93 raise ValueError("Need at least 2 clusters to calculate ICC") 

94 if df[cluster_column].isna().any(): 

95 raise ValueError(f"cluster column '{cluster_column}' contains NaN values") 

96 if df[outcome_column].isna().any(): 

97 raise ValueError(f"outcome column '{outcome_column}' contains NaN values") 

98 

99 # Note: If we need to add covariates, we can use a mixed model ICC here if the dataset is large enough. 

100 return _icc_one_way_random_intercept(df, cluster_column=cluster_column, outcome_column=outcome_column)
